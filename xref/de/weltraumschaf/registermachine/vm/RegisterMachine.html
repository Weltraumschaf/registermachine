12  package de.weltraumschaf.registermachine.vm;
13  
14  import de.weltraumschaf.commons.IO;
15  import de.weltraumschaf.registermachine.Const;
16  import de.weltraumschaf.registermachine.instr.Instruction;
17  import java.util.List;
18  import org.apache.commons.lang.StringUtils;
19  
20  public class RegisterMachine {
21      private static final int DEBUG_PAD = 26;
22  
23      private final RuntimeConfiguration config;
24      private final boolean debug;
25      private final boolean printProgramm;
26      private final IO io;
27  
28      private List<Instruction> program;
29  
30      public  RegisterMachine(final IO io) {
31          this(io, false, false);
32      }
33  
34      public  RegisterMachine(final IO io, boolean debug, boolean printProgramm) {
35          super();
36          this.io = io;
37          this.config = new RuntimeConfiguration();
38          this.debug  = debug;
39          this.printProgramm  = printProgramm;
40      }
41  
42      public void setProgram(final List<Instruction> prog) {
43          program = prog;
44      }
45  
46      public void run() {
47          final StringBuilder debugOutput = new StringBuilder();
48          config.setInstructionCounter(0);
49          debugInit(debugOutput);
50  
51          if (!program.isEmpty()) {
52              while ( config.getInstructionCounter() < program.size()) {
53                  final Instruction instruction = program.get(config.getInstructionCounter());
54                  config.incInstructionCounter();
55                  instruction.evaluate(config);
56                  debugInstruction(debugOutput, instruction);
57              }
58              io.println("HALT.");
59              io.println("");
60          }
61  
62          if (debug) {
63              io.println(String.format("Debug:%n======%n%s", debugOutput.toString()));
64          }
65  
66          if (this.printProgramm) {
67              this.printProgramm();
68          }
69      }
70  
71      public void printProgramm() {
72          io.println(String.format("Programm:%n========="));
73  
74          for (final Instruction instruction : program) {
75              io.println(instruction.toString());
76          }
77  
78          io.print(Const.NL);
79      }
80  
81      public void printConfiguration() {
82          io.println(config.toString());
83      }
84  
85      public RuntimeConfiguration getConfiguration() {
86          return this.config;
87      }
88  
89      private void debugInit(final StringBuilder debugOutput) {
90          if (debug) {
91              debugOutput.append(StringUtils.rightPad("Init", DEBUG_PAD))
92                         .append(" > ")
93                         .append(config.toString())
94                         .append(Const.NL);
95          }
96      }
97  
98      private void debugInstruction(final StringBuilder debugOutput, final Instruction instruction) {
99          if (debug) {
100             debugOutput.append(StringUtils.rightPad("Instruction '" + instruction.toString() + "'", DEBUG_PAD))
101                        .append(" > ")
102                        .append(config.toString())
103                        .append(Const.NL);
104         }
105     }

107 }
