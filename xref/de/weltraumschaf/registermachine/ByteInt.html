12  package de.weltraumschaf.registermachine;

14  /**
15   * Converts integers to array of bytes and vice versa.
16   *
17   * Converts big endian.
18   *
19   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
20   */
21  public final class ByteInt {
22  
23      /** How much bits to shift 2nd byte. */
24      private static final int SECOND_BYTE_SHIFT = 8;
25      /** How much bits to shift 3rd byte. */
26      private static final int THIRD_BYTE_SHIFT = 16;
27      /** How much bits to shift 4th byte. */
28      private static final int FOURTH_BYTE_SHIFT = 24;
29  
30      public static final int ONE_BYTE_MASK = 0xFF;
31      private static final int TWO_BYTE_MASK = 0xFFFF;
32      private static final int THREE_BYTE_MASK = 0xFFFFFF;
33      private static final int FOUR_BYTE_MASK = 0xFFFFFFFF;
34  
35      private static final int FIRST_BYTE = 0;
36      private static final int SECOND_BYTE = 1;
37      private static final int THIRD_BYTE = 2;
38      private static final int FOURTH_BYTE = 3;
39  
40      private static final int BYTES_SHORT = 2;
41      private static final int BYTES_INT = 4;
42  
43      private static final String BYTE_COUNT_ERROR_FMT = "Need %d bytes exactly!";
44  
45  
46      /**
47       * Hide constructor for pure static class.
48       */
49      private ByteInt() {
50          super();
51      }
52  
53      /**
54       * Converts 16 bit short to two byte arrays.
55       *
56       * @param value short to convert
57       * @return array with 2 bytes
58       */
59      public static byte[] bytesFromShort(final short value) {
60          final byte[] bytes = new byte[BYTES_SHORT];
61          bytes[FIRST_BYTE]  = (byte) value;
62          bytes[SECOND_BYTE] = (byte) (value >> SECOND_BYTE_SHIFT);
63          return bytes;
64      }
65  
66      /**
67       * Converts array of two bytes into 16 bit short integer.
68       *
69       * @param bytes array with exactly 2 bytes
70       * @return short integer
71       * @throws IllegalArgumentException if, the passed in array does not have exactly 2 bytes
72       */
73      public static short shortFromBytes(final byte[] bytes) {
74          if (bytes.length != BYTES_SHORT) {
75              throw new IllegalArgumentException(String.format(BYTE_COUNT_ERROR_FMT, BYTES_SHORT));
76          }
77  
78          short value = (short) (bytes[0]                         & ONE_BYTE_MASK);
79          value      |= (bytes[SECOND_BYTE] << SECOND_BYTE_SHIFT) & TWO_BYTE_MASK;
80          return value;
81      }
82  
83      /**
84       * Converts 32 bit short to four byte arrays.
85       *
86       * @param value int to convert
87       * @return array with 4 bytes
88       */
89      public static byte[] bytesFromInt(final int value) {
90          final byte[] bytes = new byte[BYTES_INT];
91          bytes[FIRST_BYTE]  = (byte) value;
92          bytes[SECOND_BYTE] = (byte) (value >> SECOND_BYTE_SHIFT);
93          bytes[THIRD_BYTE]  = (byte) (value >> THIRD_BYTE_SHIFT);
94          bytes[FOURTH_BYTE] = (byte) (value >> FOURTH_BYTE_SHIFT);
95          return bytes;
96      }
97  
98      /**
99       * Converts array of four bytes into 32 bit short integer.
100      *
101      * @param bytes array with exactly 4 bytes
102      * @return int integer
103      * @throws IllegalArgumentException if, the passed in array does not have exactly 4 bytes
104      */
105     public static int intFromBytes(final byte[] bytes) {
106         if (bytes.length != BYTES_INT) {
107             throw new IllegalArgumentException(String.format(BYTE_COUNT_ERROR_FMT, BYTES_INT));
108         }
109 
110         int value = bytes[FIRST_BYTE]                         & ONE_BYTE_MASK;
111         value    |= (bytes[SECOND_BYTE] << SECOND_BYTE_SHIFT) & TWO_BYTE_MASK;
112         value    |= (bytes[THIRD_BYTE]  << THIRD_BYTE_SHIFT)  & THREE_BYTE_MASK;
113         value    |= (bytes[FOURTH_BYTE] << FOURTH_BYTE_SHIFT) & FOUR_BYTE_MASK;
114         return value;
115     }

117 }
