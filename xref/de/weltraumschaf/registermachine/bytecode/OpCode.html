17  /**
18   * Opcodes of the byte code.
19   *
20   * General format:
21   * <pre>
22   * [ opcode 1 byte ]
23   * [ opcode 1 byte ] [ arg1 4 byte ]
24   * [ opcode 1 byte ] [ arg1 4 byte ] [ arg2 4 byte ]
25   * [ opcode 1 byte ] [ arg1 4 byte ] [ arg2 4 byte ] [ arg3 4 byte ]
26   * <pre>
27   *
28   * TODO Make result register implicitly to 0.
29   *
30   * @author sxs
31   */
32  public enum OpCode {
33  
34      /**
35       * Copy a value between registers.
36       *
37       * <pre>
38       * move SRC_REG DST_REG ; copy register SRC_REG into register DST_REG
39       * </pre>
40       */
41      MOVE(0x01, ArgCount.TWO),
42      /**
43       * Load a constant into a register.
44       *
45       * <pre>
46       * loadc DST_REG VALUE ; load integer value into register DST_REG
47       * </pre>
48       */
49      LOADC(0x02, ArgCount.TWO),
50      /**
51       * Addition operator.
52       *
53       * <pre>
54       * add OP1_REG OP2_REG ; add value from OP1_REG and OP2_REG and stores result in register 0
55       * </pre>
56       */
57      ADD(0x03, ArgCount.TWO),
58      /**
59       * Subtraction operator.
60       *
61       * <pre>
62       * sub OP1_REG OP2_REG ; subtract value from OP1_REG and OP2_REG and stores result in register 0
63       * </pre>
64       */
65      SUB(0x04, ArgCount.TWO),
66      /**
67       * Multiplication operator.
68       *
69       * <pre>
70       * mul RES_REG OP1_REG OP2_REG ; multiply value from OP1_REG and OP2_REG and stores result in RES_REG
71       * </pre>
72       */
73      MUL(0x05, ArgCount.TWO),
74      /**
75       * Division operator.
76       *
77       * <pre>
78       * div OP1_REG OP2_REG ; divide value from OP1_REG and OP2_REG and stores result in register 0
79       * </pre>
80       */
81      DIV(0x06, ArgCount.TWO),
82      /**
83       * Modulus (remainder) operator.
84       *
85       * <pre>
86       * mod OP1_REG OP2_REG ; reminder divide value from OP1_REG and OP2_REG and stores result in register 0
87       * </pre>
88       */
89      MOD(0x07, ArgCount.TWO),
90      /**
91       * Exponentiation operator.
92       *
93       * <pre>
94       * pow OP1_REG OP2_REG ; powers value from OP1_REG and OP2_REG and stores result in register 0
95       * </pre>
96       */
97      POW(0x08, ArgCount.TWO),
98      /**
99       * Unary minus operator.
100      */
101     UNM(0x09, ArgCount.ONE),
102     /**
103      * Logical NOT operator.
104      */
105     NOT(0x0a, ArgCount.ONE),
106     /**
107      * Unconditional jump.
108      */
109     JMP(0x0b, ArgCount.ONE),
110     /**
111      * Equality test.
112      */
113     EQ(0x0c, ArgCount.TWO),
114     /**
115      * Less than test Less than or equal to test.
116      */
117     LT(0x0d, ArgCount.TWO),
118     /**
119      * Less than or equals test Less than or equal to test.
120      */
121     LE(0x0e, ArgCount.TWO),
122     /**
123      *Boolean test, with conditional jump.
124      */
125     TEST(0x0f, ArgCount.TWO),
126     /**
127      *
128      */
129     PRINT(0x10, ArgCount.ONE),
130     /**
131      *
132      */
133     PRINTLN(0x11, ArgCount.ONE),
134     /**
135      * Unknown opcode wil stop execution.
136      */
137     UNKWONN(0xff);
138 
139     private static final Map<String, OpCode> MNEMONIC_LOOKUP = Maps.newHashMap();
140     private static final Map<Byte, OpCode> OPCODE_LOOKUP = Maps.newHashMap();
141 
142     static {
143         for (final OpCode code : OpCode.values()) {
144             MNEMONIC_LOOKUP.put(code.name().toLowerCase(Const.LOCALE), code);
145             OPCODE_LOOKUP.put(Byte.valueOf(code.getCode()), code);
146         }
147     }
148 
149     private final byte code;
150     private final ArgCount argCount;
151 
152     OpCode(final int code) {
153         this(code, ArgCount.NONE);
154     }
155 
156     OpCode(final int code, ArgCount argCount) {
157         this((byte) code, argCount);
158     }
159 
160     OpCode(final byte code, ArgCount argCount) {
161         this.code = code;
162         this.argCount = argCount;
163     }
164 
165     public byte getCode() {
166         return code;
167     }
168 
169     public ArgCount getArgCount() {
170         return argCount;
171     }
172 
173     @Override
174     public String toString() {
175         return String.format("%s[0x%s]", name(), toHex(code));
176     }
177 
178     public String toHex() {
179         return toHex(code);
180     }
181 
182     public static String toHex(final byte b) {
183         final String hex = Integer.toHexString(0xFF & b);
184         return hex.length() == 1
185                 ? "0" + hex
186                 : hex;
187     }
188 
189     public static OpCode lokup(final String mnemonic) {
190         if (MNEMONIC_LOOKUP.containsKey(mnemonic.toLowerCase(Const.LOCALE))) {
191             return MNEMONIC_LOOKUP.get(mnemonic.toLowerCase(Const.LOCALE));
192         }
193 
194         return UNKWONN;
195     }
196 
197     public static OpCode lookup(final byte b) {
198         if (OPCODE_LOOKUP.containsKey(Byte.valueOf(b))) {
199             return OPCODE_LOOKUP.get(Byte.valueOf(b));
200         }
201 
202         return UNKWONN;
203     }

212         }

214         public int getCount() {
