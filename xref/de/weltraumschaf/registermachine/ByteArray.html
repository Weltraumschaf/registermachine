20  public final class ByteArray {

23          super();

26      /**
27       * Not thread safe!
28       *
29       * @param in
30       * @return
31       */
32      public static byte[] toNative(final Byte[] in) {
33          final byte[] out = new byte[in.length];
34          for (int i = 0; i < in.length; ++i) {
35              out[i] = in[i];
36          }
37          return out;
38      }
39  
40      public static Byte[] toObject(final byte[] in) {
41          final Byte[] out = new Byte[in.length];
42          for (int i = 0; i < in.length; ++i) {
43              out[i] = in[i];
44          }
45          return out;
46      }
47  
48      public static byte[] convertToNativeArray(final List<Byte> bytecode) {
49          return ByteArray.toNative(bytecode.toArray(new Byte[bytecode.size()]));
50      }
